class MyList(list):
super().__init__()

print(sorted(self))
# Output: [1, 4, 2, 3, 5]

# Creating another instance of MyList
my_list2 = MyList()

# Appending integers in a different order
my_list2.append(10)
my_list2.append(7)
my_list2.append(9)
my_list2.append(8)
my_list2.append(6)

# Printing the original list
print(my_list2)  # Output: [10, 7, 9, 8, 6]

# Sorting and printing the list using the print_sorted method
my_list2.print_sorted()  # Output: [6, 7, 8, 9, 10]

# Printing the list after calling print_sorted (should remain unchanged)
print(my_list2)  # Output: [10, 7, 9, 8, 6]
